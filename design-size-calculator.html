function calculateDesignSize() {
let actualWidth =
parseFloat(document.getElementById('actualWidth').value);
let actualHeight =
parseFloat(document.getElementById('actualHeight').value);
let onScreenWidth =
parseFloat(document.getElementById('onScreenWidth').value);
let onScreenHeight =
parseFloat(document.getElementById('onScreenHeight').value);
let designElementWidth =
parseFloat(document.getElementById('designElementWidth').value);
let designElementHeight =
parseFloat(document.getElementById('designElementHeight').value);
let error = document.getElementById('error');
let output = document.getElementById('output');
let calculatedWidth =
document.getElementById('calculatedWidth');
let calculatedHeight =
document.getElementById('calculatedHeight');
Check if all fields have valid inputs
if (isNaN(actualWidth) isNaN(actualHeight)
isNaN(onScreenWidth) isNaN(onScreenHeight) isNaN(designElementWidth)
isNaN(designElementHeight)) {
error.innerText = 'Error Please enter valid numbers in
all fields.';
output.style.display = 'none';
return;
} else {
error.innerText = '';
}
Calculate ratio
let widthRatio = actualWidth onScreenWidth;
let heightRatio = actualHeight onScreenHeight;
Calculate new design element size
let newDesignElementWidth = designElementWidth widthRatio;
let newDesignElementHeight = designElementHeight
heightRatio;
Display the result
calculatedWidth.innerText = newDesignElementWidth.toFixed(2);
calculatedHeight.innerText =
newDesignElementHeight.toFixed(2);
output.style.display = 'block';
}
Unit conversion (Inches to cm and vice versa)
document.getElementById('unitInput').addEventListener('input',
function() {
let value = parseFloat(this.value);
let result = document.getElementById('conversionResult');
if (isNaN(value)) {
result.innerText = '';
return;
}
Convert inches to cm or cm to inches
let conversion = value 2.54; Conversion factor
result.innerText = `${value} inches =
${conversion.toFixed(2)} cm`;
});
